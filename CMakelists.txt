cmake_minimum_required(VERSION 3.15)
project(AgendaDeContactos)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(PkgConfig REQUIRED)
pkg_check_modules(GTKMM REQUIRED IMPORTED_TARGET gtkmm-4.0)

add_library(CapaInterfaz STATIC 
    presentacion/Interfaz.cpp
)

target_include_directories(CapaInterfaz PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})
target_link_libraries(CapaInterfaz PRIVATE PkgConfig::GTKMM)

add_executable(${PROJECT_NAME} 
    main.cpp
    presentacion/Interfaz.cpp
    negocios/TablaHash.cpp
)

add_definitions(-D_HAS_STD_BYTE=0)
add_compile_options(/wd4250)

target_link_libraries(${PROJECT_NAME} PRIVATE 
    CapaInterfaz
    PkgConfig::GTKMM
)

add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
    # 1. Crea la carpeta base 'presentacion'
    COMMAND ${CMAKE_COMMAND} -E make_directory "$<TARGET_FILE_DIR:${PROJECT_NAME}>/presentacion"

    # 2. Copia el archivo style.css
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    "${CMAKE_CURRENT_SOURCE_DIR}/presentacion/style.css"
    "$<TARGET_FILE_DIR:${PROJECT_NAME}>/presentacion/style.css"

    # 3. Copia toda la carpeta 'assets' (donde está tu contacto.png)
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    "${CMAKE_CURRENT_SOURCE_DIR}/presentacion/assets"
    "$<TARGET_FILE_DIR:${PROJECT_NAME}>/presentacion/assets"

    COMMENT "Copiando recursos (CSS e imágenes) a la carpeta del ejecutable..."
)
