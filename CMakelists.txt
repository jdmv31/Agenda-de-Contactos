cmake_minimum_required(VERSION 3.15)
project(AgendaDeContactos)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(PkgConfig REQUIRED)
pkg_check_modules(GTKMM REQUIRED IMPORTED_TARGET gtkmm-4.0)

add_library(CapaInterfaz STATIC 
    presentacion/Interfaz.cpp
)

target_include_directories(CapaInterfaz PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})
target_link_libraries(CapaInterfaz PRIVATE PkgConfig::GTKMM)

add_executable(${PROJECT_NAME} main.cpp)

add_definitions(-D_HAS_STD_BYTE=0)
add_compile_options(/wd4250)

target_link_libraries(${PROJECT_NAME} PRIVATE 
    CapaInterfaz
    PkgConfig::GTKMM
)

# ... (todo tu código anterior) ...

# 1. Creamos el comando para copiar el archivo style.css después de compilar
add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
    # Primero: Asegura que exista la carpeta 'presentacion' en el build
    COMMAND ${CMAKE_COMMAND} -E make_directory $<TARGET_FILE_DIR:${PROJECT_NAME}>/presentacion
    
    # Segundo: Copia el archivo style.css si ha cambiado
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    "${CMAKE_CURRENT_SOURCE_DIR}/presentacion/style.css"
    "$<TARGET_FILE_DIR:${PROJECT_NAME}>/presentacion/style.css"
    
    # Mensaje en consola para que sepas que lo hizo
    COMMENT "Copiando style.css a la carpeta del ejecutable..."
)
