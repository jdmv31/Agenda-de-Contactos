cmake_minimum_required(VERSION 3.15)
project(AgendaDeContactos)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(PkgConfig REQUIRED)
pkg_check_modules(GTKMM REQUIRED IMPORTED_TARGET gtkmm-4.0)

# Generación de la ruta para los datos
file(TO_CMAKE_PATH "${CMAKE_CURRENT_SOURCE_DIR}/datos" RUTA_LIMPIA)
file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/PathConfig.h" 
     "#define RUTA_DATOS \"${RUTA_LIMPIA}\"")

# --- LIBRERÍA DE INTERFAZ ---
add_library(CapaInterfaz STATIC 
    presentacion/Interfaz.cpp
)

# CORRECCIÓN 1: La librería también necesita acceso a PathConfig.h
target_include_directories(CapaInterfaz PUBLIC 
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_BINARY_DIR}
)
target_link_libraries(CapaInterfaz PRIVATE PkgConfig::GTKMM)

# --- EJECUTABLE PRINCIPAL ---
add_executable(${PROJECT_NAME} 
    main.cpp
    presentacion/Interfaz.cpp
    negocios/TablaHash.cpp
    negocios/Lista.cpp
    datos/Ficheros.cpp
)

add_definitions(-D_HAS_STD_BYTE=0)
add_compile_options(/wd4250)

target_link_libraries(${PROJECT_NAME} PRIVATE 
    CapaInterfaz
    PkgConfig::GTKMM
)

# CORRECCIÓN 2: El ejecutable necesita acceso a PathConfig.h para compilar Ficheros.cpp
target_include_directories(${PROJECT_NAME} PRIVATE 
    ${CMAKE_CURRENT_BINARY_DIR}
)

# Comandos post-compilación para recursos
add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E make_directory "$<TARGET_FILE_DIR:${PROJECT_NAME}>/presentacion"
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    "${CMAKE_CURRENT_SOURCE_DIR}/presentacion/style.css"
    "$<TARGET_FILE_DIR:${PROJECT_NAME}>/presentacion/style.css"
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    "${CMAKE_CURRENT_SOURCE_DIR}/presentacion/assets"
    "$<TARGET_FILE_DIR:${PROJECT_NAME}>/presentacion/assets"
    COMMENT "Copiando recursos (CSS e imágenes) a la carpeta del ejecutable..."
)
